PROGRAM -> PUBLIC_FUNCTION FUNCTIONS

PUBLIC_FUNCTION -> public FUNCTION
FUNCTIONS -> FUNCTION FUNCTIONS | epsilon
FUNCTION -> function name ( PARAMS ) STATEMENTS_BLOCK

PARAMS -> name MORE_PARAMS | epsilon
MORE_PARAMS -> , name MORE_PARAMS

STATEMENTS_BLOCK -> begin STATEMENTS end
STATEMENTS -> STATEMENT STATEMENTS | epsilon
STATMENT -> RETURN
            | ASSIGN
            | IF
            | WHILE

        EXPR -> EXPR + T | EXPR - T | T
EXPR -> T EXPR'
EXPR' -> + T EXPT'| - T EXPR'| epsilon
        T -> T * F | T / F | F
T -> F T'
T' -> * F T' | / F T' | epsilon
F -> ( EXPR ) | number | VARIABLE_OR_FUNCTION_CALL

VARIABLE_OR_FUNCTION_CALL -> name FUNCTION_CALL
FUNCTION_CALL -> ( FUNCTION_CALL_PARAMS ) | epsilon
FUNCTION_CALL_PARAMS -> EXPR MORE_FUNCTION_CALL_PARAMS | epsilon
MORE_FUNCTION_CALL_PARAMS -> , EXPR MORE_FUNCTION_CALL_PARAMS | epsilon

RETURN -> return EXPR ;

ASSIGN -> name = EXPR ;

IF -> if ( EXPR ) then STATEMENTS_BLOCK ELSE ;
ELSE -> else STATEMENTS_BLOCK | epsilon

WHILE -> while ( EXPR ) STATEMENTS_BLOCK ;
